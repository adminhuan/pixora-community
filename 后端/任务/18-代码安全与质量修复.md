```typescript
async toggleLike(userId: string, targetType: ContentType, targetId: string) {
  return prisma.$transaction(async (tx) => {
    const existing = await tx.like.findUnique({
      where: { userId_targetType_targetId: { userId, targetType, targetId } },
    });

    if (existing) {
      // 取消点赞
      await tx.like.delete({ where: { id: existing.id } });

      await this.updateLikeCount(tx, targetType, targetId, 1);

    }
  });
}
```

```typescript
async create(data: CreateCommentDto) {

    await tx.comment.update({ where: { id }, data: { status: 'deleted' } });
    await this.incrementCommentCount(tx, comment.targetType, comment.targetId, -1);

    if (comment.parentId) {
      await tx.comment.update({
        where: { id: comment.parentId },
        data: { replyCount: { decrement: 1 } },
      });
    }
  });
```

```typescript
async search(keyword: string, type?: string, page = 1, limit = 20) {
  const where = {
    OR: [
      { title: { contains: keyword, mode: 'insensitive' as const } },
      { content: { contains: keyword, mode: 'insensitive' as const } },
    ],
  };

  // 根据 type 查询不同的表
  const results: any[] = [];

  if (!type || type === 'post') {
    const posts = await prisma.post.findMany({
      where: { ...where, status: 'published' },
      select: { id: true, title: true, createdAt: true, authorId: true },
      take: limit,
      skip: (page - 1) * limit,
    });
    results.push(...posts.map(p => ({ ...p, type: 'post' })));
  }

  // 博客、问题、项目、代码片段同理...

  return results;
}
```

### M-05 关注/取关操作使用事务

**修复方式：**
```typescript
async follow(followerId: string, followingId: string) {
  if (followerId === followingId) {
    throw new AppError('不能关注自己', 400);
  }

  return prisma.$transaction(async (tx) => {
    const existing = await tx.follow.findUnique({
      where: { followerId_followingId: { followerId, followingId } },
    });
    if (existing) throw new AppError('已关注', 400);

    await tx.follow.create({ data: { followerId, followingId } });
    await tx.user.update({ where: { id: followerId }, data: { followingCount: { increment: 1 } } });
    await tx.user.update({ where: { id: followingId }, data: { followersCount: { increment: 1 } } });
  });
}

async unfollow(followerId: string, followingId: string) {
  return prisma.$transaction(async (tx) => {
    const existing = await tx.follow.findUnique({
      where: { followerId_followingId: { followerId, followingId } },
    });
    if (!existing) throw new AppError('未关注', 400);

    await tx.follow.delete({ where: { id: existing.id } });
    await tx.user.update({ where: { id: followerId }, data: { followingCount: { decrement: 1 } } });
    await tx.user.update({ where: { id: followingId }, data: { followersCount: { decrement: 1 } } });
  });
}
```

```typescript
import { CorsOptions } from 'cors';

const corsOptions: CorsOptions = {
  origin: (origin, callback) => {
    const allowedOrigins = process.env.CORS_ORIGINS
      ? process.env.CORS_ORIGINS.split(',').map(s => s.trim())
      : [];

    if (process.env.NODE_ENV === 'development') {
      // 开发环境放宽限制
      callback(null, true);
      return;
    }

    if (!origin || allowedOrigins.length === 0) {
      callback(new Error('CORS: 生产环境必须配置 CORS_ORIGINS'));
      return;
    }

    if (allowedOrigins.includes(origin)) {
      callback(null, true);
    } else {
      callback(new Error(`CORS: ${origin} 不在允许列表中`));
    }
  },
  credentials: true,
};

export default corsOptions;
```
